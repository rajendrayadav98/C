#include int main() { int n; printf("Enter the number of elements: "); scanf("%d", &n); int arr[n]; int oddSum = 0, evenSum = 0; printf("Enter the elements of the array:\n"); for (int i = 0; i < n; i++) { scanf("%d", &arr[i]); } for (int i = 0; i < n; i++) { if (arr[i] % 2 == 0) { evenSum += arr[i]; // Add to even sum if the number is even } else { oddSum += arr[i]; // Add to odd sum if the number is odd } } printf("Sum of even numbers: %d\n", evenSum); printf("Sum of odd numbers: %d\n", oddSum); return 0; }
